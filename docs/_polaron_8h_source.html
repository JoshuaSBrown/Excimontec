27  void calculateExecutionTime(const double prefactor,const double localization,const double distance,const double E_delta,Simulation* sim_ptr){
28  // Calculates hopping using the Miller-Abrahams model
29  double rate = prefactor*exp(-2.0*localization*distance);
30  if(E_delta>0){
31  rate *= exp(-E_delta/(Utils::K_b*sim_ptr->getTemp()));
32  }
33  Event::calculateExecutionTime(rate,sim_ptr);
34  }
35  void calculateExecutionTime(const double prefactor,const double localization,const double distance,const double E_delta,const double reorganization, Simulation* sim_ptr){
36  // Calculates hopping using the Marcus model
37  double rate = (prefactor/sqrt(4.0*Utils::Pi*reorganization*Utils::K_b*sim_ptr->getTemp()))*exp(-2.0*localization*distance)*exp(-Utils::intpow(reorganization+E_delta,2)/(4.0*reorganization*Utils::K_b*sim_ptr->getTemp()));
38  Event::calculateExecutionTime(rate,sim_ptr);
39  }
